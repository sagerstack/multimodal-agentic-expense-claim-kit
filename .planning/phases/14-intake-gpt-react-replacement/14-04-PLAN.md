```python
_BUTTON_INTERRUPT_KINDS = {"field_confirmation", "submit_confirmation"}


def _deriveUiKind(kind: str) -> str:
    return "buttons" if kind in _BUTTON_INTERRUPT_KINDS else "text"


def _deriveButtonOptions(kind: str) -> list[dict]:
    if kind not in _BUTTON_INTERRUPT_KINDS:
        return []
    return [
        {"label": "Yes", "value": "yes"},
        {"label": "No", "value": "no"},
    ]
```

```python
uiKind = _deriveUiKind(kind)
```

```python
# Check for interrupt via graph state
try:
    if finalState and finalState.next:
        for task in finalState.tasks:
            if hasattr(task, "interrupts") and task.interrupts:
                payload = task.interrupts[0].value
                if isinstance(payload, dict):
                    contextMessage = str(payload.get("contextMessage", "") or "").strip()
                    if contextMessage:
                        try:
                            template = templates.get_template("partials/message_bubble.html")
                            contextHtml = template.render(
                                content=contextMessage,
                                isAi=True,
                                confidenceScores=None,
                                violations=None,
                                timestamp=datetime.now(
                                    ZoneInfo("Asia/Singapore")
                                ).strftime("%-I:%M %p"),
                            )
                        except Exception:
                            contextHtml = f'<div class="ai-message">{contextMessage}</div>'
                        yield ServerSentEvent(raw_data=contextHtml, event=SseEvent.MESSAGE)

                    # Dispatch on uiKind for interrupt-prompt rendering.
                    uiKind = str(payload.get("uiKind", "text"))
                    question = (
                        payload.get("question", str(payload))
                        if isinstance(payload, dict)
                        else str(payload)
                    )
                    if uiKind == "buttons":
                        options = payload.get("options") or []
                        try:
                            buttonTemplate = templates.get_template(
                                "partials/interrupt_buttons.html"

                                question=question,
                                options=options,
                            )
                        except Exception:
                            # Fallback to plain text if template rendering fails.
                            interruptHtml = str(question)
                        yield ServerSentEvent(
                            raw_data=interruptHtml, event=SseEvent.INTERRUPT
                        )
                    else:
                        yield ServerSentEvent(
                            raw_data=str(question), event=SseEvent.INTERRUPT
                        )
                else:
                    # Non-dict payload (defensive): yield as plain text.
                    yield ServerSentEvent(raw_data=str(payload), event=SseEvent.INTERRUPT)
                return
except Exception as e:
    logEvent(
```

**IMPORTANT — working directory and template path resolution:**

The test `test_interruptButtonsPartialRendersYesNoButtons` calls `templates.get_template("partials/interrupt_buttons.html")`. Jinja2's `FileSystemLoader` (as configured in `agentic_claims.web.templating`) resolves template paths RELATIVE to the loader's search path — typically the `templates/` directory at the repo root. If pytest is not run from the repo root (e.g., run from `/tmp` or a subdirectory), the template lookup will fail.

To guard against this, add an explicit path assertion at the TOP of the test and use `pathlib.Path` to resolve the template file relative to the test file itself:

```python
from pathlib import Path

_REPO_ROOT = Path(__file__).resolve().parent.parent  # tests/ -> repo root
_TEMPLATE_PATH = _REPO_ROOT / "templates" / "partials" / "interrupt_buttons.html"


def test_interruptButtonsPartialRendersYesNoButtons():
    """Partial template renders both Yes and No buttons with correct form fields.

    This test resolves the template path relative to __file__ so it is robust
    to pytest being invoked from any working directory. It also asserts the
    template file exists on disk before attempting to load it via Jinja2.
    """
    assert _TEMPLATE_PATH.exists(), (
        f"Template file not found at {_TEMPLATE_PATH} — the test must be able to resolve "
        f"templates/ relative to the tests directory. Did Plan 14-04 Task 1b run?"
    )

    from agentic_claims.web.templating import templates

    template = templates.get_template("partials/interrupt_buttons.html")
    html = template.render(
        question="Do these extracted details look correct?",
        options=[
            {"label": "Yes", "value": "yes"},
            {"label": "No", "value": "no"},
        ],
    )
    assert "Do these extracted details look correct?" in html
    assert ">Yes</button>" in html
    assert ">No</button>" in html
    assert 'name="button_value"' in html
    assert 'value="yes"' in html
    assert 'value="no"' in html
    assert 'hx-post="/chat/message"' in html
```
