       def _mergeSubgraphResult(state: dict, result: dict) -> dict:
           """Propagate a whitelisted subset of subgraph result back to outer state.

           See Plan 13-06 Task 1b for the authoritative whitelist and semantics.
           """
           merged: dict = {}

           # messages â€” delta form (see Plan 06 Task 1b: pick one form and document)
